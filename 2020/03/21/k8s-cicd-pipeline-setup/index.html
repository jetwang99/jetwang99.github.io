Setup nexus private registry

generate ssl certificate and restart nexus

export NEXUS_HOST=your_hostname
export NEXUS_DOMAIN=your_domain_name.com
export NEXUS_IP_ADDRESS=192.168.5.12
export NEXUS_SSL_PORT=18444

keytool -genkeypair -keystore keystore.jks -storepass 5St00g3s -keypass 5St00g3s -alias jetty -keyalg RSA -keysize 2048 -validity 5000 -dname "CN=$NEXUS_DOMAIN, OU=Example, O=Sonatype, L=Unspecified, ST=Unspecified, C=US" -ext "SAN=DNS:$NEXUS_DOMAIN,IP:$NEXUS_IP_ADDRESS" -ext "BC=ca:true"

openssl req -newkey rsa:4096 -nodes -sha256 -keyout ${nexus_home}/etc/ssl/nexus.key -x509 -days 3650 -out ${nexus_home}/etc/ssl/nexus.cert

ls ${nexus_home}/etc/ssl/
keystore.jks  nexus.cert  nexus.key

etc/nexus-default.properties

application-port-ssl=8444
nexus-args=${jetty.etc}/jetty.xml,${jetty.etc}/jetty-http.xml,${jetty.etc}/jetty-https.xml,${jetty.etc}/jetty-requestlog.xml

etc/nexus.properties

application-port-ssl=18444

restart nexus service

${nexus_home}/bin/nexus restart

create nexus repository and set Repository Connectors in nexus UI

#hosted type repository for docker push

setup ssl trust in docker client node(not always necessary)

#remove old cert
cd /etc/pki/ca-trust/source/anchors
rm –f your_domain_name.crt

#client trust
export DOMAIN_NAME=your_domain_name
export TCP_PORT=18444
openssl s_client -connect $DOMAIN_NAME:$TCP_PORT -showcerts </dev/null 2>/dev/null | openssl x509 -outform PEM | tee /etc/pki/ca-trust/source/anchors/$DOMAIN_NAME.crt

update-ca-trust

configure /etc/docker/daemon.json and restart docker service

{
    "insecure-registries" : ["your_domain_name:18444","your_domain_name:18445"]
}

restart docker service and login

/bin/systemctl restart docker.service

#test docker login
docker login your_domain_name:18444
docker login your_domain_name:18445

Create ci cd pipeline in k8s (non-docker)

pipeline {
    agent { label "jenkins-jenkins-slave" } // jenkins slave with maven sonar-scanner tools

    parameters {
        // build param
        string(name: 'PKG_VERSION', defaultValue: '1.0.0', description: '固定值，不需要填写')
        string(name: 'BRANCH', defaultValue: 'master', description: '固定值，不需要填写')
        string(name: 'SONAR_PROJECT_NAME', defaultValue: 'sonar_project_name', description: '固定值，不需要填写')
        ...
        choice(name: 'ENABLE_SONAR', choices: ['N', 'Y'], description: '是否开启SONAR扫描')
        choice(name: 'DEPLOY_QA', choices: ['N', 'Y'], description: '是否发布QA环境\n\n')
    }

    environment {
        SONAR_HOST = "http://nginx_url/sonarqube"
        SONAR_TOKEN = ""

        // nexus for artifacts upload
        NEXUS_VERSION = "nexus3"
        NEXUS_PROTOCOL = "http"
        NEXUS_URL = ""
        NEXUS_REPOSITORY = "snapshot"
        NEXUS_GROUP = ""
        NEXUS_ARTIFACT_ID = ""
        NEXUS_CREDENTIAL_ID = ""

        releasePackageDirPath = "RELEASE/patch/jenkins/${PILOT_PROJECT}"
    }

    options {
        buildDiscarder(logRotator(numToKeepStr: '5'))
        preserveStashes()
        { skipDefaultCheckout() }
    }

    stages {
        stage("Code Checkout") {
            steps { ... }
        }

        stage('Build') {
            steps {
                script{
                    try {
                        script {
                            try {
                                sh 'mvn package -f project_name/pom.xml -Dmaven.test.skip=true'
                            } finally { }
                        }
                    } catch(error) {
                        echo "First build failed, let's retry if accepted"
                        retry(2) { // same code as above }
                    }
                }
            }
        }

        stage('Assemble & Test') {
            parallel {
                stage("Assemble") {
                    steps { sh "sh package.sh" }
                }
            }
        }

        stage('Scan & Upload') {
            parallel {
                stage('Archive Artifacts') {
                    steps {
                        archiveArtifacts "project_name-ms-assembly-${PKG_VERSION}-SNAPSHOT.zip"

                        sh '''
                        [ ! -d ${HOME}/${releasePackageDirPath} ] && mkdir -p ${HOME}/${releasePackageDirPath}
                        cp -a project_name-ms-assembly-${PKG_VERSION}-SNAPSHOT.zip ${HOME}/${releasePackageDirPath}
                        '''
                    }
                }
                // stage("Upload Artifacts") {
                //  steps {
                //      nexusArtifactUploader \
                //          credentialsId: "${NEXUS_CREDENTIAL_ID}", \
                //          groupId: "${NEXUS_GROUP}", \
                //          nexusUrl: "${NEXUS_URL}", \
                //          nexusVersion: "${NEXUS_VERSION}", \
                //          protocol: "${NEXUS_PROTOCOL}", \
                //          repository: "${NEXUS_REPOSITORY}", \
                //          version: "${PKG_VERSION}", \
                //          artifacts: [
                //              [artifactId: "${NEXUS_ARTIFACT_ID}", \
                //              classifier: '', \
                //              file: "project_name-ms-assembly-${PKG_VERSION}-SNAPSHOT.zip", \
                //              type: 'zip']
                //          ]
                //  }
                // }
                stage("SonarQube Scan") {
                    steps {
                        script {
                            if (env.ENABLE_SONAR == 'Y') {
                                withSonarQubeEnv('dev_sonarqube') { // configured in jenkins global setting
                                    sh '''sonar-scanner -Dsonar.host.url=${SONAR_HOST} \
                                        -Dsonar.sourceEncoding=UTF-8 \
                                        -Dsonar.sources=. \
                                        -Dsonar.language=java \
                                        -Dsonar.login=${SONAR_TOKEN} \
                                        -Dsonar.projectVersion=${BRANCH} \
                                        -Dsonar.java.binaries=. \
                                        -Dsonar.projectKey=${SONAR_PROJECT_NAME} \
                                        -Dsonar.java.libraries=. \
                                        -Dsonar.projectName=${SONAR_PROJECT_NAME}
                                    '''
                                }
                                sleep(120)
                                timeout(time: 1, unit: 'MINUTES') {
                                    def qg = waitForQualityGate()
                                    if (qg.status != 'SUCCESS') {
                                        echo "Pipeline aborted due to quality gate failure: ${qg.status}"
                                        // error "Pipeline aborted due to quality gate failure: ${qg.status}"
                                    }
                                }
                            } else { }
                        }
                    }
                }
            }
        }

        stage('Deploy DEV') {
            agent { label "jenkins-slave-cap" } // ruby capistrano slave, deploy dev by default
            steps {
                script {
                    // sh "cap -T"
                    sh '''#!/bin/bash -ex

                    cd  ${HOME}/${releasePackageDirPath}
                    md5sum project_name-ms-assembly-${PKG_VERSION}-SNAPSHOT.zip > project_name-ms-assembly-${PKG_VERSION}-SNAPSHOT.zip.md5sum
                    [ -d project_name ] && rm -rf project_name
                    unzip -o project_name-ms-assembly-${PKG_VERSION}-SNAPSHOT.zip

                    cd /data/relmgt
                    cap  ${ENVTYPE_DEV} patch:upload CHG=${HOME}/${releasePackageDirPath}/project_name-ms-assembly-${PKG_VERSION}-SNAPSHOT.zip PROJECT=${PILOT_PROJECT}
                    cap  ${ENVTYPE_DEV} patch:upload CHG=${HOME}/${releasePackageDirPath}/project_name-ms-assembly-${PKG_VERSION}-SNAPSHOT.zip.md5sum PROJECT=${PILOT_PROJECT}

                    # other deploy logic here
                    '''
                }
            }
        }

        stage('Deploy QA') {
            agent { label "jenkins-slave-cap" }
            when {
                environment name: 'DEPLOY_QA', value: 'Y'
            }
            steps {
                script { // refer dev deploy script }
            }
        }
    }
}

Create ci cd pipeline in k8s (build docker image and push to private registry)

stage("Build Docker Image") {
    steps {
        script {
            def app

            stage('Build image') {
                withDockerRegistry(credentialsId: "${IMAGE_NEXUS_CREDENTIAL_ID}", url: "http://${REPO}/v1") {
                    app = docker.build("${REPO_PUSH}/project_name/app_name-service:${env.DOMAIN}", "--label project_name/app_name-service --no-cache --pull --force-rm --build-arg REPO=${REPO} --build-arg DOMAIN=${DOMAIN} --build-arg GROUP_ID=${DOMAIN} --build-arg LAUNCHER_VERSION=${LAUNCHER_VERSION} --build-arg PKG_VERSION=${IMAGE_VERSION} --build-arg NEXUS_REPO=${NEXUS_REPO} .")
                }
            }

            stage('Push image') {
                withDockerRegistry(credentialsId: "${IMAGE_NEXUS_CREDENTIAL_ID}", url: "http://${REPO_PUSH}/v1") {
                    app.push("${env.IMAGE_VERSION}-${env.BUILD_NUMBER}")
                    app.push("${env.IMAGE_VERSION}-latest")
                }
            }
        }
    }
}
